Algorithm q1(A, B):
    Input: arrays A and B each containing n >= 1 integers (they both contain the same number of integers)
    Output: <unknown>

    int c = 0;
    for (int i = 0; i < n; i++) {
        int s = 0;
        for (int j = 0; j < n; j++) {
            s = s + A[0];
            for (int k = 1; k < j; k++) {
                s = s + A[k];
            }
        }
        if (B[i] == s) {
            c++;
        }
    }
    return c;
